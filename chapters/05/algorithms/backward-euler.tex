$t = 0$\;
$[X] = [X_0]$\;


\While{$t<T_{\max}$}{
	estimate $[X_{new}] = [X] + h\cdot\vec{F}(t, [X])$\;
	$t = t+h$\;
	$[X] = [X] + h\cdot\vec{f}(t, [X_{new}])$\;
}
