\Input: $M$, $p(x)$ and $q(x)$\;

\Output: a random variable distributed according to $p$\;

$i = 1$\;
\Repeat{}{
	sample $x^{(i)}\sim q$, $u\sim norm(0,1)$\;
	\If{$u>\frac{p(x^{(i)})}{Mq(X^i)}$}{
		$i = i + 1$\;
	}
}
