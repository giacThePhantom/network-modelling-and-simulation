\caption{\protect\TitleFunction{}}
\label{algo:mh}

\Input: $N$, $r$ and $q(x)$\;

\Output: a random variable distributed according to $p$\;

guess $x^{(0)}$\;

\ForEach{$i = 0,\dots, N-1$}{
	sample $x^{(*)}\sim q(\cdot|x^{(i)})$ and $u\sim norm(0,1)$\;
	\If{$u<\min\{1, r_H(x,x^*)\}$}{
		$x^{(i+1)} = x^{(*)}$\;
	}
	\Else{
		$x^{(i+1)} = x^{(i)}$\;
	}
}
